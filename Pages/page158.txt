struct RefCV {
     void   func()  &              {}
     void   func()  &&             {}
     void   func()  const&         {}
     void   func()  const&&        {}
     void   func()  volatile&      {}
     void   func()  volatile&&     {}
     void func() const volatile& {}
     void func() const volatile&& {}
};
